        // can safely assume that the user wants to
        // skip the prefix and begin at the root.
        if (path.indexOf('./') == 0) {
            return path;
        }

        } else {
            path = p.join(prefix, path);
        }

            .replace(/\/\//g, '/')
            .replace(p.join(prefix, prefix), prefix);
